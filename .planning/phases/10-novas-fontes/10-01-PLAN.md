    @require_admin
    def api_admin_source_stats():
        with get_db() as conn:
            c = conn.cursor()
            c.execute('''
                SELECT source, COUNT(*) AS total
                FROM leads
                WHERE captured_at >= NOW() - INTERVAL '30 days'
                  AND source IS NOT NULL
                GROUP BY source
                ORDER BY total DESC
            ''')
            rows = c.fetchall()
        return jsonify([{'source': r[0], 'count': r[1]} for r in rows])
